\begin{algorithm}                                        
\caption{High-Level Description of E3}          
\label{e3_high_level}                                             
\begin{algorithmic}
    \State set all states as unknown
    \Loop
        \While{in unknown state}
            \State perform balanced wandering 
        \EndWhile
        \State perform value iteration on two MPD:s (see text above)
        \If{chance of ending up in unknown state is greater than threshold}
            \For{$i=1$ {\bf to} horizon time}
                \State perform action of the policy from the first value iteration
                \If{in unknown state}
                    \State {\bf break}
                \EndIf
            \EndFor
        \Else
            \For{$i=1$ {\bf to} horizon time}
                \State perform action of the policy given by the second value iteration
                \If{in unknown state}
                    \State {\bf break}
                \EndIf
            \EndFor
        \EndIf
    \EndLoop
\end{algorithmic}
\end{algorithm}
